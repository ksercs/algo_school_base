Битовое "и" =  "&"
Битовое "Или" = "|"
Битовый сдвиг вправо = ">>"
Битовый сдвиг влево = "<<"
Битовое "!" = "~"
Битовый "кcор" = "^"

int getBit(int a, int b)
{
  return a&(1<<b);
}

int setBit(int a, int b, int v)
{
  if (v==1) {return a|(1<<b);}
  else 	    {return a&(~0)^(1<<b);}
}
